**Week 3-4: First Synthesizer**
```swift
// Implement simple sine wave synthesizer
// File: Core/Audio/Synthesizers/SimpleSynth.swift

import AVFoundation

@MainActor
final class SimpleSynth: ObservableObject {
    private var phase: Double = 0
    private var frequency: Double = 440 // A4

    func renderAudio(frameCount: AVAudioFrameCount) -> [Float] {
        var buffer = [Float](repeating: 0, count: Int(frameCount))
        let phaseIncrement = frequency / sampleRate * 2.0 * .pi

        for i in 0..<Int(frameCount) {
            buffer[i] = Float(sin(phase))
            phase += phaseIncrement
            if phase > 2.0 * .pi {
                phase -= 2.0 * .pi
            }
        }

        return buffer
    }
}
```
