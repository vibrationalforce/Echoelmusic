class WorkflowGenerator

        Project project;

        {
            // Template: Hip-Hop Beat Production

            project.addTrack("Snare", TrackType::Audio);
            project.addTrack("Hi-Hat", TrackType::Audio);
            project.addTrack("808 Bass", TrackType::MIDI);
            project.addTrack("Melody", TrackType::MIDI);
            project.addTrack("Vocals", TrackType::Audio);

            // Routing
            project.addBus("Drums", {0, 1, 2}); // Kick, Snare, Hi-Hat
            project.addBus("Master", {3, 4, 5});

            // Default plugins
            project.tracks[0].addPlugin("EQ", {boost: "60Hz", cut: "200Hz"});
            project.tracks[5].addPlugin("De-Esser");
            project.addMasterPlugin("Limiter", {ceiling: -0.3dB, LUFS: -14.0});

            // Tempo + Time Signature
            project.setTempo(90); // Typical Hip-Hop BPM
            project.setTimeSignature(4, 4);
        }

        return project;
    }

    // Save as reusable template
    void saveAsTemplate(const Project& project, const String& name);
};
